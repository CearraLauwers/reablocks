# Design Tokens Blocks
A set of components to use in Storybook to display design tokens. It provides
the following block components:

- Typography Block
- Color Block
- Spacing Block
- Border Block
- Shadow Block
- Palette Block
- Components Block
- Gradients Block
- Icons Block ( icons not included )

## Example
Here is an example Storybook story that uses the Design Tokens Blocks.

```tsx
import React from 'react';
import {
  TypographyBlocks,
  ColorBlocks,
  SpacingBlocks,
  PaletteBlocks,
  BorderBlocks,
  ShadowBlocks,
  ComponentBlocks,
  DesignTokensProvider,
  lightTheme,
  IconBlocks
} from 'reablocks';

export default {
  title: 'General/Design Tokens',
  component: DesignTokensProvider,
  decorators: [
    Story => (
      <DesignTokensProvider value={lightTheme}>
        <div style={{ width: '90vw' }}>
          <Story />
        </div>
      </DesignTokensProvider>
    )
  ]
};

export const Colors = () => (
  <ColorBlocks />
);

export const Palettes = () => (
  <PaletteBlocks />
);

export const Typography = () => (
  <TypographyBlocks />
);

export const Spacings = () => (
  <SpacingBlocks />
);

export const Borders = () => (
  <BorderBlocks />
);

export const Gradients = () => (
  <GradientBlocks />
);

export const Shadows = () => (
  <ShadowBlocks />
);

export const Components = () => (
  <ComponentBlocks />
);

const icons = [
  {
    name: 'favorite',
    src: favoriteIconSrc
  },
   {
    name: 'favorite',
    component: FavoriteIconComponent
  }
];

export const Icons = () => (
  <IconBlocks icons={icons} />
);
````

NOTE: This example uses the `DesignTokensProvider` in the story decorator,
but you can also use it in the `preview.js` file to wrap all stories.
